class AbsErrorErrFunc(teller.nonconformist.nc.RegressionErrFunc): 
146class AbsErrorErrFunc(RegressionErrFunc):
147    """Calculates absolute error nonconformity for regression problems.
148
149    For each correct output in ``y``, nonconformity is defined as
150
151    .. math::
152        | y_i - \hat{y}_i |
153    """
154
155    def __init__(self):
156        super(AbsErrorErrFunc, self).__init__()
157
158    def apply(self, prediction, y):
159        return np.abs(prediction - y)
160
161    def apply_inverse(self, nc, significance):
162        nc = np.sort(nc)[::-1]
163        border = int(np.floor(significance * (nc.size + 1))) - 1
164        # TODO: should probably warn against too few calibration examples
165        border = min(max(border, 0), nc.size - 1)
166        return np.vstack([nc[border], nc[border]])

class QuantileRegErrFunc(teller.nonconformist.nc.RegressionErrFunc): 
207class QuantileRegErrFunc(RegressionErrFunc):
208    """Calculates conformalized quantile regression error.
209
210    For each correct output in ``y``, nonconformity is defined as
211
212    .. math::
213        max{\hat{q}_low - y, y - \hat{q}_high}
214
215    """
216
217    def __init__(self):
218        super(QuantileRegErrFunc, self).__init__()
219
220    def apply(self, prediction, y):
221        y_lower = prediction[:, 0]
222        y_upper = prediction[:, -1]
223        error_low = y_lower - y
224        error_high = y - y_upper
225        err = np.maximum(error_high, error_low)
226        return err
227
228    def apply_inverse(self, nc, significance):
229        nc = np.sort(nc, 0)
230        index = int(np.ceil((1 - significance) * (nc.shape[0] + 1))) - 1
231        index = min(max(index, 0), nc.shape[0] - 1)
232        return np.vstack([nc[index], nc[index]])

Base class for all estimators in scikit-learn.

Inheriting from this class provides default implementations of:

  • setting and getting parameters used by GridSearchCV and friends;
  • textual and HTML representation displayed in terminals and IDEs;
  • estimator serialization;
  • parameters validation;
  • data validation;
  • feature names validation.

Read more in the :ref:User Guide <rolling_your_own_estimator>.

Notes

All estimators should specify all the parameters that can be set at the class level in their __init__ as explicit keyword arguments (no *args or **kwargs).

class RegressorNc(teller.nonconformist.nc.BaseModelNc): 
514class RegressorNc(BaseModelNc):
515    """Nonconformity scorer using an underlying regression model.
516
517    Parameters
518    ----------
519    model : RegressorAdapter
520        Underlying regression model used for calculating nonconformity scores.
521
522    err_func : RegressionErrFunc
523        Error function object.
524
525    normalizer : BaseScorer
526        Normalization model.
527
528    beta : float
529        Normalization smoothing parameter. As the beta-value increases,
530        the normalized nonconformity function approaches a non-normalized
531        equivalent.
532
533    Attributes
534    ----------
535    model : RegressorAdapter
536        Underlying model object.
537
538    err_func : RegressionErrFunc
539        Scorer function used to calculate nonconformity scores.
540
541    See also
542    --------
543    ProbEstClassifierNc, NormalizedRegressorNc
544    """
545
546    def __init__(
547        self, model, err_func=AbsErrorErrFunc(), normalizer=None, beta=1e-6
548    ):
549        super(RegressorNc, self).__init__(model, err_func, normalizer, beta)
550
551    def predict(self, x, nc, significance=None):
552        """Constructs prediction intervals for a set of test examples.
553
554        Predicts the output of each test pattern using the underlying model,
555        and applies the (partial) inverse nonconformity function to each
556        prediction, resulting in a prediction interval for each test pattern.
557
558        Parameters
559        ----------
560        x : numpy array of shape [n_samples, n_features]
561            Inputs of patters for which to predict output values.
562
563        significance : float
564            Significance level (maximum allowed error rate) of predictions.
565            Should be a float between 0 and 1. If ``None``, then intervals for
566            all significance levels (0.01, 0.02, ..., 0.99) are output in a
567            3d-matrix.
568
569        Returns
570        -------
571        p : numpy array of shape [n_samples, 2] or [n_samples, 2, 99]
572            If significance is ``None``, then p contains the interval (minimum
573            and maximum boundaries) for each test pattern, and each significance
574            level (0.01, 0.02, ..., 0.99). If significance is a float between
575            0 and 1, then p contains the prediction intervals (minimum and
576            maximum	boundaries) for the set of test patterns at the chosen
577            significance level.
578        """
579        n_test = x.shape[0]
580        prediction = self.model.predict(x)
581        if self.normalizer is not None:
582            norm = self.normalizer.score(x) + self.beta
583        else:
584            norm = np.ones(n_test)
585
586        if significance:
587            intervals = np.zeros((x.shape[0], 2))
588            err_dist = self.err_func.apply_inverse(nc, significance)
589            err_dist = np.hstack([err_dist] * n_test)
590            if prediction.ndim > 1:  # CQR
591                intervals[:, 0] = prediction[:, 0] - err_dist[0, :]
592                intervals[:, 1] = prediction[:, -1] + err_dist[1, :]
593            else:  # regular conformal prediction
594                err_dist *= norm
595                intervals[:, 0] = prediction - err_dist[0, :]
596                intervals[:, 1] = prediction + err_dist[1, :]
597
598            return intervals
599        else:  # Not tested for CQR
600            significance = np.arange(0.01, 1.0, 0.01)
601            intervals = np.zeros((x.shape[0], 2, significance.size))
602
603            for i, s in enumerate(significance):
604                err_dist = self.err_func.apply_inverse(nc, s)
605                err_dist = np.hstack([err_dist] * n_test)
606                err_dist *= norm
607
608                intervals[:, 0, i] = prediction - err_dist[0, :]
609                intervals[:, 1, i] = prediction + err_dist[0, :]
610
611            return intervals
